API Integration Snippet

Copy and paste this Javascript fetch request into your website or application to connect your form to Reputifly.

javascript 
// Reputifly Nexus API - Direct Email Integration
// NOTE: Make sure your root website domain (e.g., reputifly.org) is whitelisted in your dashboard!
// Once the root is added, it automatically works for www., http://, https://, and all subpages.

const payload = {
    apikey: "YOUR_API_KEY_HERE",
    _send_to: "customer@example.com", 
    _auto_reply_subject: "Your Custom Subject",
    _auto_reply_content: "
Hello!
Your custom message here.
"
};

fetch('https://asia-southeast1-b2b-software.cloudfunctions.net/reputifly', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'Accept': 'application/json'
    },
    body: JSON.stringify(payload)
})
.then(response => response.json())
.then(data => console.log('Nexus Response:', data))
.catch(error => console.error('Nexus Error:', error));
// Code will be generated here
